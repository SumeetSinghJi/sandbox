GITHUB COMMANDS

TERMINOLOGY
upstream: the original repo e.g, vedic-lang
origin: your fork of upstream e.g, SumeetSinghJi/vedic-lang


STEP 1 - FORK REPO
# e.g here click CREATE FORK: https://github.com/vedic-lang/vedic

STEP 2 - CLONE REPO
git clone https://github.com/SumeetSinghJi/vedic
cd vedic
git remote add upstream https://github.com/vedic-lang/vedic
git fetch upstream
git merge upstream/main
git reset --hard upstream/main

STEP 3 - CREATE BRANCH
git checkout -b SUMEETS_BRANCH

STEP OPTIONAL IF FILE CREATED/EDITED ON WINDOWS (e.g. for uploading Brew formulas)
# removes carraige returns
sed -i '' 's/\r//' /Users/sumeetsingh/Documents/homebrew-vedic/vedic.rb
# give all write access
chmod +x vedic.rb

STEP 4 - COMMIT/PUSH
git add .
git commit -m "Add your descriptive commit message here"
git push origin SUMEETS_BRANCH

STEP 5 - PULL REQUEST
# Go to your forked repo and click COMPARE PULL REQUEST and accept

STEP 6 - RESET/SYNC/CLEANUP FORK
git remote add upstream https://github.com/vedic-lang/vedic
git fetch upstream
git merge upstream/main
git reset --hard upstream/main

BREW FORMULA COMMIT WORKFLOW

# step 1 - fork repo https://github.com/Homebrew/homebrew-core
brew tap --force homebrew/core
cd "$(brew --repository homebrew/core)"
cp ~/Documents/sandbox/vedic.rb Formula/v
cd Formula/v
HOMEBREW_VERBOSE=1 HOMEBREW_NO_AUTO_UPDATE=1 brew reinstall --build-from-source vedic.rb


# RUN TESTS
brew test vedic
brew audit --strict --online vedic
brew audit --new vedic
brew style --fix vedic


# RUN CLEANUP
brew uninstall vedic
brew cleanup vedic
brew cleanup --prune=all
rm -rf $(brew --cache)/*
rm -rf $(brew --repository)/Library/Logs/Homebrew/*

# COMMIT + PUSH + PR (requires 2 new lines in commit message)
git remote add SumeetSinghJi https://github.com/SumeetSinghJi/homebrew-core.git
git checkout -b 2.0.6-vedic-formula origin/master
